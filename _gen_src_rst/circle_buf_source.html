  3module circle_buf #(
  4     parameter dw=16,
  5     parameter aw=13,
  6     parameter stat_w=16, // Width of buffer statistics
  7     // For each half of the double-buffered memory, the default is to use a read of the last
  8     // memory location as acknowledgement that the buffer read is complete,
  9     // and read cycles need the stb_out signal set high to register.
 10     // Set this parameter to 0 to disable that feature, in which case you
 11     // need to construct the stb_out signal as a simple explicit flip command.
 12     parameter auto_flip = 1
 13) (
 14     // source side
 15     input          iclk,
 16     input [dw-1:0] d_in,
 17     input          stb_in,          // d_in is valid
 18     input          boundary,        // between blocks of input strobes
 19     input          stop,            // single-cycle
 20                                     // assume stop happens a programmable number of samples
 21                                     // after a fault event, and we will save 1024 samples
 22                                     // before the stop signal
 23     output         buf_sync,        // single-cycle when buffer starts/ends
 24     output         buf_transferred, // single-cycle when a buffer has been
 25                                     // handed over for reading;
 26                                     // one cycle delayed from buf_sync
 27
 28     // readout side
 29     input                 oclk,
 30     output                enable,
 31     input  [aw-1:0]       read_addr, // nominally 8192 locations
 32     output [dw-1:0]       d_out,
 33     input                 stb_out,
 34     output [stat_w-1:0]   buf_count, // number of full buffer writes
 35     output [aw-1:0]       buf_stat2, // last valid location
 36     output [stat_w-1:0]   buf_stat,  // includes fault bit, and (if set) the last valid location
 37     output [aw+4:0]       debug_stat // {stb_in, boundary, btest, wbank, rbank, wr_addr}
 38);

 40`define MIN(a,b) a < b ? a : b

 42// parameterized to improve testability
 43
 44// 8192 words of 16 bits, double buffered
 45// maybe subdivided into 1024 time samples of 4 RF waveforms,
 46// each with an I and Q component
 47
 48// readout side state
 49reg rbank=0;  // really complement
 50
 51// source side control logic
 52// Flow control is opposite that in decay_buf: the pacing happens
 53// from the readout side
 54wire flag_return;   // buffer request from readout side
 55reg [aw-1:0] write_addr=0, save_addr=0, save_addr0=0;
 56reg pend=0, run=1, wbank=0, flag_return_x=0;
 57wire change_req = wbank ^ flag_return_x;
 58wire end_write_addr = &write_addr;
 59reg record_type=1;
 60reg boundary_ok=1;
 61reg [1:0] done_read = 2'b0;
 62wire stop_write=pend & boundary;
 63wire eval_done_read = stb_in & boundary_ok & end_write_addr;
 64wire eval_block=eval_done_read & change_req;
 65reg buff_wrap=0;
 66reg buf_transferred_r=0;
 67wire btest= boundary | ( stb_in & end_write_addr );
 68assign buf_transferred = buf_transferred_r;
 69always @(posedge iclk) begin
 70     flag_return_x <= flag_return;  // Clock domain crossing
 71     if (eval_done_read) done_read[wbank] <= change_req;
 72     //if (boundary|(stb_in&end_write_addr)) boundary_ok <= boundary;
 73     if (btest) boundary_ok <= boundary;
 74     if (stb_in & boundary_ok) write_addr <= write_addr+1; //wbank==rbank ? 0 : write_addr+1;
 75     if (eval_block) begin
 76             run <= 1;
 77             wbank <= ~wbank;
 78             record_type <= run;  // fault (0) vs. comfort display (1)
 79             save_addr0 <= run ? {aw{1'b0}} : save_addr;
 80     end
 81     buf_transferred_r <= eval_block;
 82     if ((stop & run)| boundary) pend <= stop & run;  // ignore double stops
 83     if (stop_write) begin
 84             run <= 0;
 85             save_addr <= write_addr;
 86             buff_wrap <= ~done_read[wbank];
 87     end
 88end
 89wire flag_send=wbank;  // says "I want to write bank foo"
 90assign debug_stat={stb_in,boundary,btest,wbank,rbank,write_addr};
 91// Handshake means "OK, I won't read bank foo"
 92
 93// readout side control logic
 94reg flag_send_x=0;
 95wire [aw-1:0] read0_addr = read_addr;  // cut down to current width
 96wire end_read_addr = &read0_addr;
 97assign enable = ~flag_send_x ^ rbank;
 98wire flip_buffer = stb_out & enable & (end_read_addr | ~auto_flip);
 99always @(posedge oclk) begin
100     flag_send_x <= flag_send;  // Clock domain crossing
101     if (flip_buffer) rbank <= ~rbank;
102end
103assign flag_return = rbank;
104
105// in iclk domain
106wire [stat_w-2-1:0] save_addr0_ext = save_addr0[`MIN(stat_w-2,aw)-1:0]; // truncate or pad MSBs
107assign buf_stat = {record_type, buff_wrap, save_addr0_ext};
108
109assign buf_stat2 = save_addr0;
110wire write_en= stb_in & boundary_ok & run;
111
112// Count of how many buffers we have acquired
113reg [stat_w-1:0] buf_count_r=0;
114always @(posedge iclk) if (eval_done_read) buf_count_r <= buf_count_r+1;
115assign buf_count=buf_count_r;
116assign buf_sync=eval_done_read;
117
118// data path is simply a dual-port RAM
119dpram #(.aw(aw+1), .dw(dw)) cbuf(.clka(iclk), .clkb(oclk),
120     .addra({wbank,write_addr}), .dina(d_in), .wena(write_en),
121     .addrb({~rbank,read0_addr}), .doutb(d_out)
122);
123
124endmodule
