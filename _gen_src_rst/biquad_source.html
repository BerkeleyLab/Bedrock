  1// Biquad IIR filter
  2//
  3// y(t) = u(t)*b0 + u(t-1)*b1 + u(t-2)*b2 + y(t-1)*(-a1) + y(t-2)*(-a2)
  4//
  5// Minimizes latency by computing all but u(t)*b0 at end of previous sample.
  6//
  7// Minimizes resource consumption by using internal
  8// DSP registers for between-sample storage.
  9//
 10// Writing a coefficient holds the filter in reset until address 7 is written.
 11// This allows usage of a consistent set of coefficients.
 12//
 13module biquad #(
 14    parameter DATA_WIDTH        = 28,
 15    parameter DATA_COUNT        = 1,
 16    parameter COEFFICIENT_WIDTH = 25,
 17    parameter DEBUG             = "false"
 18) (
 19    input                         sysClk,
 20    input                         sysCoefficientStrobe,
 21    input                   [2:0] sysCoefficientAddress,
 22    input [COEFFICIENT_WIDTH-1:0] sysCoefficientValue,
 23
 24    input                                                         dataClk,
 25    (*mark_debug=DEBUG*) input      [(DATA_COUNT*DATA_WIDTH)-1:0] S_TDATA,
 26    (*mark_debug=DEBUG*) input                                    S_TVALID,
 27    (*mark_debug=DEBUG*) output reg                               S_TREADY,
 28    (*mark_debug=DEBUG*) output reg [(DATA_COUNT*DATA_WIDTH)-1:0] M_TDATA,
 29    (*mark_debug=DEBUG*) output reg                               M_TVALID,
 30    (*mark_debug=DEBUG*) input                                    M_TREADY
 31);
 32
 33localparam MAC_WIDEN = 4;
 34localparam MAC_WIDTH = DATA_WIDTH + COEFFICIENT_WIDTH + MAC_WIDEN;
 35
 36// Coefficient dual-port RAM 0:b0, 1:b1, 2:b2, 3:-a2, 4:-a1
 37// Coefficient range [-2,2) -- i.e. two bits to the left of the binary point
 38reg [COEFFICIENT_WIDTH-1:0] coefficientRAM [0:4], coefficientRAMq;
 39reg sysReset = 1;
 40always @(posedge sysClk) begin
 41    if (sysCoefficientStrobe) begin
 42        if (sysCoefficientAddress <= 4) begin
 43            coefficientRAM[sysCoefficientAddress] <= sysCoefficientValue;
 44            sysReset <= 1;
 45        end else if (sysCoefficientAddress == 7) begin
 46            sysReset <= 0;
 47        end
 48    end
 49end
 50
 51// I/O history
 52(*mark_debug=DEBUG*) reg [(DATA_COUNT*DATA_WIDTH)-1:0] u, uOld = 0, yOld = 0;
 53
 54// MAC parameter input multiplexer
 55wire [(DATA_COUNT*DATA_WIDTH)-1:0] parameterMux = (state == 1) ? u :
 56                                                  (state == 2) ? u :
 57                                                  (state == 3) ? uOld :
 58                                                  (state == 4) ? yOld : M_TDATA;
 59
 60// Computation state machine
 61(* ASYNC_REG="TRUE" *) reg reset_m, reset;
 62reg [2:0] state = 0;
 63reg enMAC = 0, ldMAC = 0;
 64
 65always @(posedge dataClk) begin
 66    coefficientRAMq <= coefficientRAM[state];
 67end
 68
 69always @(posedge dataClk) begin
 70    reset_m <= sysReset;
 71    reset   <= reset_m;
 72    if (reset) begin
 73        state <= 0;
 74        u <= 0;
 75        uOld <= 0;
 76        yOld <= 0;
 77        S_TREADY <= 0;
 78        M_TVALID <= 0;
 79    end
 80    else begin
 81        case (state)
 82        0: begin
 83            if (S_TVALID && S_TREADY) begin
 84                u <= S_TDATA;
 85                S_TREADY <= 0;
 86                enMAC <= 1;
 87                state <= 1;
 88            end
 89            else begin
 90                S_TREADY <= 1;
 91            end
 92        end
 93        1: begin // MAC inputs: u(t), b0
 94                 // Multiplier inputs: y(t-1), -a1
 95                 // Accumulator input: y(t-2)*-a2
 96                 // Clip input: u(t-2)*b2 + u(t-1)*b1
 97                 // M_TDATA: u(t-1)*b1
 98            state <= 2;
 99        end
100        2: begin // MAC inputs: u(t-1), b1 for next cycle
101                 // Multiplier inputs: u(t), b0
102                 // Accumulator inputx: y(t-1)*-a1
103                 // Clip input: y(t-2)*-a2 + u(t-2)*b2 + u(t-1)*b1
104                 // M_TDATA: u(t-2)*b2 + u(t-1)*b1
105            ldMAC <= 1;
106            state <= 3;
107        end
108        3: begin // MAC inputs: u(t-2), b2 for next cycle
109                 // Multiplier inputs: u(t-1), b1 for next cycle
110                 // Accumulator input: u(t)*b0
111                 // Clip input: y(t-1)*a1 + y(t-2)*-a2 + u(t-2)*b2 + u(t-1)*b1
112                 // M_TDATA: y(t-2)*-a2 + u(t-2)*b2 + u(t-1)*b1
113                 // sload = 1
114            ldMAC <= 0;
115            uOld <= u;
116            state <= 4;
117        end
118        4: begin // MAC inputs: y(t-2), -a2 for next cycle
119                 // Multiplier inputs: u(t-2), b2 for next cycle
120                 // Accumulator input: u(t-1)*b1 for next cycle
121                 // Clip input: u(t)*b0+y(t-1)*a1+y(t-2)*-a2+u(t-2)*b2+u(t-1)*b1
122                 // M_TDATA: y(t-1)*a1+y(t-2)*-a2+u(t-2)*b2+u(t-1)*b1
123                 // sload_reg = 1
124            M_TVALID <= 1;
125            state <= 5;
126        end
127        5: begin // MAC inputs: y(t-1), -a1 for next cycle
128                 // Multiplier inputs: y(t-2), -a2 for next cycle
129                 // Accumulator input: u(t-2), b2 for next cycle
130                 // Clip input: u(t-1)*b1 for next cycle
131                 // M_TDATA: u(t)*b0+y(t-1)*a1+y(t-2)*-a2+u(t-2)*b2+u(t-1)*b1
132                 // M_TVALID = 1
133            enMAC <= 0;
134            yOld <= M_TDATA;
135            if (M_TREADY) begin
136                M_TVALID <= 0;
137                S_TREADY <= 1;
138                state <= 0;
139            end
140        end
141        default: begin
142            enMAC <= 0;
143            ldMAC <= 0;
144            S_TREADY <= 0;
145            M_TVALID <= 0;
146            state <= 0;
147        end
148        endcase
149    end
150end
151
152///////////////////////////////////////////////////////////////////////////////
153// Per-lane computation
154genvar i;
155generate
156for (i = 0 ; i < DATA_COUNT ; i = i + 1) begin
157    // Instantiate multiply-accumulate module
158    // Module doesn't provide a reset port so fake one by enabling
159    // the module in 'load' mode with coefficients all 0.
160    wire [MAC_WIDTH-1:0] accum_out;
161    macc # (.SIZEA(DATA_WIDTH),
162            .SIZEB(COEFFICIENT_WIDTH),
163            .SIZEOUT(MAC_WIDTH))
164      macc_i (
165        .clk(dataClk),
166        .ce(reset || enMAC),
167        .sload(reset || ldMAC),
168        .a(reset ? {DATA_WIDTH{1'b0}} : parameterMux[i*DATA_WIDTH+:DATA_WIDTH]),
169        .b(coefficientRAMq),
170        .accum_out(accum_out));
171
172    // Clip accumulated result
173    // The '-2' on the input width and input bit selection accounts
174    // for the fact that the coefficient range is [-2,2).
175    wire [DATA_WIDTH-1:0] accum_out_clipped;
176    reduceWidth #(.IWIDTH(MAC_WIDTH-(COEFFICIENT_WIDTH-2)),
177                  .OWIDTH(DATA_WIDTH))
178      clipMAC (.I(accum_out[MAC_WIDTH-1:COEFFICIENT_WIDTH-2]),
179               .O(accum_out_clipped));
180    always @(posedge dataClk) begin
181        M_TDATA[i*DATA_WIDTH+:DATA_WIDTH] <= accum_out_clipped;
182    end
183end
184endgenerate
185endmodule

187///////////////////////////////////////////////////////////////////////////////
188// Multiply-accumulate unit
189// Template from Vivado
190module macc #(
191    parameter SIZEA   = 25,
192              SIZEB   = 28,
193              SIZEOUT = 55
194) (
195    input clk,
196    input ce,
197    input sload,
198    input signed    [SIZEA-1:0] a,
199    input signed    [SIZEB-1:0] b,
200    output signed [SIZEOUT-1:0] accum_out
201);
202
203// Declare registers for intermediate values
204reg signed       [SIZEA-1:0] a_reg;
205reg signed       [SIZEB-1:0] b_reg;
206reg                          sload_reg;
207reg signed [SIZEA+SIZEB-1:0] mult_reg;
208reg signed     [SIZEOUT-1:0] adder_out, old_result;
209
210always @(sload_reg or adder_out)
211begin
212 if (sload_reg)
213    old_result <= 0;
214 else
215  // 'sload' is now and opens the accumulation loop.
216  // The accumulator takes the next multiplier output
217  // in the same cycle.
218    old_result <= adder_out;
219end
220
221always @(posedge clk)
222 if (ce)
223  begin
224    a_reg     <= a;
225    b_reg     <= b;
226    mult_reg  <= a_reg * b_reg;
227    sload_reg <= sload;
228    // Store accumulation result into a register
229    adder_out <= old_result + mult_reg;
230 end
231
232// Output accumulation result
233assign accum_out = adder_out;
234
235endmodule
