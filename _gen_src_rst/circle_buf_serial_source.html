  3/** CIRCLE_BUF_SERIAL **
  4    Double-buffered circular buffer wrapper with channel selector.
  5    This module takes a conveyor belt carrying multiple channels as an input
  6    and writes the channels selected by chan_mask into the Circular Buffer.
  7    Circular buffer can be read out at a different clock rate through the local bus.
  8
  9          +------------------------+
 10          | +-----+    +--------+  |
 11  sr_in+---->     |    |        <-----+ rd_addr/val
 12          | |fchan+----> Circle |  |
 13  mask +----> sel |    | Buf    |  |
 14          | |     |    |        +-----> d_out
 15          | +-----+    +--------+  |
 16          +------------------------+
 17*/
 18
 19module circle_buf_serial #(
 20   // Channel selector parameters
 21   parameter n_chan=12,
 22   parameter lsb_mask=0,      // lsb_mask=0: chan_mask is LEFT-to-RIGHT; MSB=CH0
 23                              // lsb_mask=1: chan_mask is RIGHT-to-LEFT; LSB=CH0
 24
 25   // Circular Buffer parameters
 26   parameter buf_dw=16,
 27   parameter buf_aw=13,
 28   parameter buf_stat_w=16,
 29   parameter buf_auto_flip=1) // auto_flip=1: Double buffers will be flipped when
 30                              //              last read address is reached
 31                              // auto_flip=0: Buffers must be explicitly flipped by
 32                              //              using stb_out as a pulse and not a strobe
 33(
 34   // Incoming stream
 35   input                    iclk,
 36   input [buf_dw-1:0]       sr_in, // Conveyor belt carrying n_chan channels
 37   input                    sr_stb,
 38
 39   // Channel selector controls
 40   input [n_chan-1:0]       chan_mask, // Bitmask of channels to record. See lsb_mask parameter
 41   // Selected waveform data in iclk domain
 42   output                   wave_gate,
 43   output                   wave_dval,
 44   output                   [buf_dw-1:0] wave_data,
 45
 46   // Circular Buffer control and statistics
 47   // all of these signals are also in the iclk domain
 48   output                   buf_sync,        // single-cycle when buffer starts/ends
 49   output                   buf_transferred, // single-cycle when a buffer has been
 50                                             // handed over for reading;
 51                                             // one cycle delayed from buf_sync
 52   input                    buf_stop,        // single-cycle - interrupts cbuf writing
 53   output [buf_stat_w-1:0]  buf_count,
 54   output [buf_aw-1:0]      buf_stat2,       // includes fault bit
 55   output [buf_stat_w-1:0]  buf_stat,        // includes fault bit, and (if set) the last valid location
 56   output [buf_aw+4:0]      debug_stat,      // {stb_in, boundary, btest, wbank, rbank, wr_addr}
 57
 58   // Circular Buffer data readout
 59   input                    oclk,
 60   input                    stb_out,
 61   output                   enable,
 62   input  [buf_aw-1:0]      read_addr, // nominally 8192 locations
 63   output [buf_dw-1:0]      d_out
 64);
 65
 66   // ------
 67   // Interleaved channel selector
 68   // ------
 69   wire              fchan_time_error; // For debug only - will pulse if gate-per-trigger ratio is broken
 70   wire              wave_trig;
 71   assign            wave_dval = ~wave_trig;
 72
 73   fchan_subset #(
 74      .KEEP_OLD (lsb_mask),
 75      .a_dw     (buf_dw),
 76      .o_dw     (buf_dw),
 77      .len      (n_chan))
 78   i_fchan_subset (
 79      .clk      (iclk),
 80      .keep     (chan_mask),
 81      .a_data   (sr_in),
 82      .a_gate   (sr_stb),
 83      .a_trig   (~sr_stb),
 84      .o_data   (wave_data),
 85      .o_gate   (wave_gate),
 86      .o_trig   (wave_trig),
 87      .time_err (fchan_time_error)
 88   );
 89
 90`ifdef SIMULATE
 91   always @(negedge iclk) if (fchan_time_error) begin
 92      $display("ERROR: Gate-per-Trigger ratio violated in fchan_selector.");
 93      $finish;
 94   end
 95`endif
 96
 97   // ------
 98   // Double-buffered circular buffer
 99   // ------
100
101   circle_buf #(
102      .aw        (buf_aw),
103      .dw        (buf_dw),
104      .auto_flip (buf_auto_flip))
105   i_circle_buf (
106      .iclk            (iclk),
107      .d_in            (wave_data),
108      .stb_in          (wave_gate),
109      .boundary        (wave_trig),
110      .stop            (buf_stop),
111      .buf_sync        (buf_sync),
112      .buf_transferred (buf_transferred),
113      .oclk            (oclk),
114      .enable          (enable),
115      .read_addr       (read_addr),
116      .d_out           (d_out),
117      .stb_out         (stb_out),
118      .buf_count       (buf_count),
119      .buf_stat        (buf_stat),
120      .debug_stat      (debug_stat),
121      .buf_stat2       (buf_stat2)
122   );
