 2// Synthesizes to 17 slices at 155 MHz in XC3Sxxx-4 using XST-10.1i
 3
 4// Transform a raw phase difference (-pi to pi) into a control signal
 5// for a PLL.  Uses internal state to generate the right full-scale
 6// DC signal when the frequencies are mismatched.  In the final,
 7// locked-at-zero-phase state, the output equals the input.
 8
 9// Subtle API change from old pdetect: when strobe_in is not set,
10// the output follows the input exactly.
11
12// Yet another API change: new input "reset", only used when strobe_in
13// is set, resets the state machine to unwound.
14
15module pdetect(
16     input clk,
17     input [w-1:0] ang_in,
18     input strobe_in,
19     input reset,
20     output reg [w-1:0] ang_out,
21     output reg strobe_out
22);
23parameter w=17;
24
25// coding is important, see usage of next bits below
26reg [1:0] state=0;
27`define S_LINEAR 0
28`define S_CLIP_P 2
29`define S_CLIP_N 3
30
31initial ang_out=0;
32initial strobe_out=0;
33
34reg [1:0] prev_quad=0;
35wire [1:0] quad = ang_in[w-1:w-2];
36wire trans_pn = (prev_quad==2'b01) & (quad==2'b10);
37wire trans_np = (prev_quad==2'b10) & (quad==2'b01);
38
39reg [1:0] next=0;
40always @(*) begin
41     next=state;
42     if (trans_pn & (state==`S_LINEAR)) next=`S_CLIP_P;
43     if (trans_np & (state==`S_LINEAR)) next=`S_CLIP_N;
44     if (trans_pn & (state==`S_CLIP_N)) next=`S_LINEAR;
45     if (trans_np & (state==`S_CLIP_P)) next=`S_LINEAR;
46     if (reset) next=`S_LINEAR;
47end
48
49wire [w-1:0] clipv = {next[0],{w-1{~next[0]}}};
50always @(posedge clk) begin
51     if (strobe_in) begin
52             prev_quad <= quad;
53             state <= next;
54     end
55     ang_out <= (next[1] & strobe_in & ~reset) ? clipv : ang_in;
56     strobe_out <= strobe_in;
57end
58
59endmodule
