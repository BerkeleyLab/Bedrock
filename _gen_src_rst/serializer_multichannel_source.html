 4    Conveyor belt generator which takes a generic number of input vectors (flattened)

24`define FILL_BIT 1'bx

26`define FILL_BIT 1'b0

29module serializer_multichannel #(
30   parameter n_chan = 8, // Number of channels to serialize
31   parameter dw = 16,
32   parameter l_to_r=1) // l_to_r=1: Channel shifting starts with CH0 (default)
33                       // l_to_r=0: Channel shifting starts with last CH
34(
35
36   input                 clk,
37   input                 sample_in, // Sampling signal which determines when to push inputs to belt
38   input [n_chan*dw-1:0] data_in,   // Flattened array of unprocessed data streams
39   output                gate_out,
40   output [dw-1:0]       stream_out // Serialized stream of channels. Default order is CH0 first (l_to_r=1)
41);
42   wire [dw-1:0] shift_chain[n_chan:0]; // +1 for feed-out
43   wire          shift_gate[n_chan:0]; // +1 for feed-out
44   wire [dw-1:0] sr_in[n_chan-1:0];
45
46
47   assign shift_chain[0] = {dw{`FILL_BIT}}; // initialize shift-chain
48   assign shift_gate[0]  = 1'b0;
49
50   genvar ch_id;
51   generate for (ch_id=0; ch_id < n_chan; ch_id=ch_id+1) begin : g_serializer
52      if (l_to_r == 0) // CH0 first
53         assign sr_in[ch_id] = data_in[(n_chan-ch_id)*dw-1: (n_chan-ch_id-1)*dw];
54      else
55         assign sr_in[ch_id] = data_in[(ch_id+1)*dw-1: ch_id*dw];
56
57      serialize #(
58         .dwi(dw))
59      i_serialize_ch
60      (
61         .clk        (clk),
62         .samp       (sample_in),
63         .data_in    (sr_in[ch_id]),
64         .stream_in  (shift_chain[n_chan - ch_id - 1]),
65         .stream_out (shift_chain[n_chan - ch_id]),
66         .gate_in    (shift_gate[n_chan - ch_id -1]),
67         .gate_out   (shift_gate[n_chan - ch_id]),
68         .strobe_out () // Open - unused
69      );
70   end endgenerate
71
72   assign gate_out   = shift_gate[n_chan];
73   assign stream_out = shift_chain[n_chan];
74
75endmodule
