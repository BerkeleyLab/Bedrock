  1/** COMPLEX_FREQ_WRAP **
  2    Wraps around complex_freq.v and adds COMB (CCFILT) portion
  3    of upstream CIC filter and conveyor belt channel selection.
  4
  5    Assumes following conveyor belt ordering:
  6    {I,Q} * {Field, Forward, Reverse, Fiber, Drive, spare}
  7*/
  8
  9module complex_freq_wrap #(
 10   parameter n_chan     = 12,
 11   parameter sr_wi      = 40, // Conveyor belt data width
 12   parameter shift_base = 4,
 13   parameter refcnt_w   = 17
 14) (
 15   input              clk,         // single clock domain
 16   input  [1:0]       channel_sel, // 0 - Field, 1 - Forward, 2 - Reverse, 3 - IQ_Fiber (from PRC)
 17   input              sample_wave,
 18   input  [3:0]       wave_shift,
 19   input              sr_valid,
 20   input  [sr_wi-1:0] sr_data,
 21   output signed [refcnt_w-1:0] reg_freq,
 22   output             reg_freq_valid,
 23   output [16:0]      reg_amp_max,
 24   output [16:0]      reg_amp_min,
 25   output             reg_updated,
 26   // Additional outputs
 27   output [23:0]      avg_power,
 28   output             avg_power_strobe,
 29   output             reg_timing_err
 30);
 31   localparam CCFILT_OUTW = 20;
 32   localparam CFREQ_INW   = 18;
 33   localparam FIELD_CHID = 0, FWD_CHID = 1, REV_CHID = 2, IQFIB_CHID = 3;
 34
 35   wire signed [CCFILT_OUTW-1:0] cc_result;
 36   wire cc_strobe;
 37   wire signed [CFREQ_INW-1:0] cfreq_data;
 38   wire signed [CFREQ_INW-1:0] cfreq_din;
 39   wire cfreq_valid;
 40
 41   reg  [n_chan-1: 0] fchan_mask, fchan_mask_r;
 42   wire fchan_time_err, cfreq_time_err;
 43
 44   ccfilt #(
 45      .use_hb     (0),
 46      .use_delay  (0),
 47      .dsr_len    (n_chan),
 48      .dw         (sr_wi),
 49      .outw       (CCFILT_OUTW),
 50      .shift_base (shift_base))
 51   ccfilt(
 52      .clk        (clk),
 53      .reset      (1'b0),
 54      .sr_in      (sr_data),
 55      .sr_valid   (sr_valid & sample_wave),
 56      .shift      (wave_shift),
 57      .result     (cc_result),
 58      .strobe     (cc_strobe));
 59
 60   // Truncate LSBs - Assumes that CCFILT_OUTW >= CFREQ_INW
 61   assign cfreq_data = cc_result[CCFILT_OUTW-1:CCFILT_OUTW-CFREQ_INW];
 62
 63   // Channel selection
 64   always @(channel_sel) begin
 65      fchan_mask = 0;
 66      case (channel_sel)
 67         FIELD_CHID : fchan_mask[1:0] = 2'b11;
 68         FWD_CHID   : fchan_mask[3:2] = 2'b11;
 69         REV_CHID   : fchan_mask[5:4] = 2'b11;
 70         IQFIB_CHID : fchan_mask[7:6] = 2'b11;
 71      endcase
 72   end
 73
 74   // Latch onto new mask between strobes
 75   always @(posedge clk) if (cc_strobe == 0) fchan_mask_r <= fchan_mask;
 76
 77   fchan_subset #(
 78      .KEEP_OLD (1),
 79      .a_dw     (CFREQ_INW),
 80      .o_dw     (CFREQ_INW),
 81      .len      (n_chan))
 82   i_fchan_subset (
 83      .clk      (clk),
 84      .reset    (1'b0),
 85      .keep     (fchan_mask_r),
 86      .a_data   (cfreq_data),
 87      .a_gate   (cc_strobe),
 88      .a_trig   (~cc_strobe),
 89      .o_data   (cfreq_din),
 90      .o_gate   (cfreq_valid),
 91      .o_trig   (),
 92      .time_err (fchan_time_err)
 93   );
 94
 95   wire [23:0] square_sum_out;
 96   wire square_sum_valid;
 97   complex_freq #(
 98      .refcnt_w(refcnt_w))
 99   i_complex_freq (
100      .clk        (clk),
101      .sdata      (cfreq_din),
102      .sgate      (cfreq_valid),
103      .freq       (reg_freq),
104      .freq_valid (reg_freq_valid),
105      .amp_max    (reg_amp_max),
106      .amp_min    (reg_amp_min),
107      .updated    (reg_updated),
108      .timing_err (cfreq_time_err),
109      .square_sum_out   (square_sum_out),
110      .square_sum_valid (square_sum_valid)
111   );
112
113   // First-order CIC filter of square_sum
114   localparam square_sum_ex=15;  // average and decimate by 2^square_sum_ex
115   cic_simple_us #(.dw(24), .ex(square_sum_ex)) cic(.clk(clk),
116      .data_in(square_sum_out), .data_in_gate(square_sum_valid),
117      .roll(1'b0),
118      .data_out(avg_power), .data_out_gate(avg_power_strobe)
119   );
120
121   assign reg_timing_err = cfreq_time_err | fchan_time_err;
122
123endmodule
