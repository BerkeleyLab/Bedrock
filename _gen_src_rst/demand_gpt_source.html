 3// timing error logic for simple single-input module that
 4// requires a fixed gates-per-trig (gpt).
 5module demand_gpt(input clk, input gate, input trig, output time_err);
 6
 7parameter gpt=16;
 8
 9reg time_err_r=0;
10reg gate_check=0;
11reg [8:0] count=0;  // XXX generous, but not general
12
13always @(posedge clk) begin
14        gate_check <= gate;
15     count <= count + gate;
16     if (trig && gate_check) begin
17             time_err_r <= (count+gate) != gpt;
18             count <= 0;
19     end
20end
21
22assign time_err = time_err_r;
23
24endmodule
