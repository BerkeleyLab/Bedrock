  1// permute np ports, each dw bits wide, using rl levels of 2-way mux
  2//  np must be 2**rl
  3//
  4// Networking equipment with similar topology is known as a Banyan switch,
  5// so I'll use that name.  A non-folded Banyan switch actually has two
  6// switches of this type back-to-back; I can leave off the second half
  7// because in my application the output channels are fungible.
  8//
  9// The intended use case is to route 8 x ADC data to 8 x data sinks,
 10// in any of the following configurations:
 11//  8 parallel  :   1 combination    x  1 time state
 12//  4 parallel  :  70 combinations   x  2 time states
 13//  2 parallel  :  28 combinations   x  4 time states
 14//  1 at a time :   8 possibilities  x  8 time states
 15// for 1*1 + 70*2 + 28*4 + 8*8 = 317 useful configurations
 16//
 17// Represent the channel selection configuration with an 8-bit mask of
 18// which channels to look at.  Only 1+70+28+8 = 107 of 256 possible masks
 19// are valid, since it's invalid to have 0, 3, 5, 6, or 7 bits set.
 20// Configuration also includes 3 bits of time_state.
 21//
 22// time_state and mask_in are processed at full bandwidth.
 23// There are rl+1 cycles of latency from a change of these
 24// control inputs to the new data_out permutation appearing.
 25// There are only rl cycles of latency to mask_out.
 26//
 27// The implementation is done with recursive instantiation of itself.
 28// The capability of doing recursion in Verilog is explicitly acknowledged
 29// and blessed by the Verilog-2005 standard, and it works without problem
 30// for me in Icarus Verilog, Verilator, Xilinx XST, and Xilinx Vivado.
 31//
 32// An N=8 dw=16 instantiation in Kintex speed grade 2 seems capable of
 33// running at ridiculous clock rates (over 550 MHz?), consuming on the
 34// order of 430 LUT.  The data muxes alone ought to use 384 LUT outputs.
 35// Even a lowly Spartan-6 should manage 160 MHz, using about 226 LUT
 36// (437 LUT outputs).
 37//
 38// Used successfully on hardware since 2016.
 39
 40module banyan #(
 41     parameter dw=16,  // data width
 42     parameter np=8,   // number of ports, must be a power of 2
 43     parameter rl=3    // number of routing layers == log_2(np)
 44) (
 45     input clk,
 46     input [rl-1:0] time_state,
 47     input [np-1:0] mask_in,
 48     input [dw*np-1:0] data_in,
 49     output [np-1:0] mask_out,   // for DPRAM write-enable
 50     output [dw*np-1:0] data_out
 51);
 52
 53localparam M = np/2;   // number of swap-boxes
 54wire two_or_more;  // use balance mode, when more than one bit of mask is set
 55two_set #(.dw(np)) two_set(.d(mask_in), .two(two_or_more));
 56
 57wire [M-1:0] mask_upper = mask_in[2*M-1:M];
 58wire [M-1:0] mask_lower = mask_in[  M-1:0];
 59wire any_lower = |mask_lower;
 60
 61// The below statement creates a Ripple-Carry chain of xors with the initial
 62// bit being fed from the right (a 1'b0). The in and out refer to the circuit
 63// verilator lint_off UNOPTFLAT
 64wire [M:0] imbalance_in;
 65wire [M-1:0] imbalance_out = imbalance_in ^ mask_upper ^ mask_lower;
 66assign imbalance_in = {imbalance_out,1'b0};
 67// verilator lint_on UNOPTFLAT
 68
 69// Priority set sources to Lower sinks when in Balance.
 70// If currently NOT imbalanced, and lower is 0 and upper 1, Flip and route to Lower
 71// If currently     imbalanced, and lower is 1 and upper 0, Flip and route to Upper
 72// You can at most flip M/2 channels
 73wire [M-1:0] flip_bal = imbalance_in & ~mask_upper &  mask_lower
 74                     | ~imbalance_in &  mask_upper & ~mask_lower;
 75
 76// If only single source; Alternate routing to different sink based on time
 77wire [M-1:0] flip_deal = { M {~any_lower ^ time_state[rl-1]}};
 78wire [M-1:0] flip_ctl = two_or_more ? flip_bal : flip_deal;
 79
 80reg [M-1:0] out_mask_upper=0, out_mask_lower=0, data_flip=0;
 81reg [rl-1:0] recurse_time_state=0;  // upper bit will be ignored
 82always @(posedge clk) begin
 83     out_mask_upper <= flip_ctl & mask_lower | ~flip_ctl & mask_upper;
 84     out_mask_lower <= flip_ctl & mask_upper | ~flip_ctl & mask_lower;
 85     recurse_time_state <= time_state;
 86     data_flip <= flip_ctl;  // pipeline
 87end
 88
 89// Data goes through one cycle after control
 90reg [dw*np-1:0] stage=0;
 91genvar jx;
 92generate for (jx=0; jx<M; jx=jx+1) begin: ss
 93always @(posedge clk) begin
 94     stage[(jx+0+1)*dw-1:(jx+0)*dw] <= data_flip[jx] ? data_in[(jx+M+1)*dw-1:(jx+M)*dw] : data_in[(jx+0+1)*dw-1:(jx+0)*dw];
 95     stage[(jx+M+1)*dw-1:(jx+M)*dw] <= data_flip[jx] ? data_in[(jx+0+1)*dw-1:(jx+0)*dw] : data_in[(jx+M+1)*dw-1:(jx+M)*dw];
 96end
 97end endgenerate
 98
 99// Recursive step
100generate if (rl > 1) begin: halvsies
101     banyan #(.rl(rl-1), .dw(dw), .np(np/2)) top(.clk(clk), .time_state(recurse_time_state[rl-2:0]), .mask_in(out_mask_upper),
102             .data_in(stage[dw*np-1:dw*np/2]), .data_out(data_out[dw*np-1:dw*np/2]), .mask_out(mask_out[2*M-1:M]));
103     banyan #(.rl(rl-1), .dw(dw), .np(np/2)) bot(.clk(clk), .time_state(recurse_time_state[rl-2:0]), .mask_in(out_mask_lower),
104             .data_in(stage[dw*np/2-1:    0]), .data_out(data_out[dw*np/2-1:    0]), .mask_out(mask_out[  M-1:0]));
105end else begin: passthrough
106     // base of recursion
107     assign data_out = stage;
108     assign mask_out = {out_mask_upper,out_mask_lower};
109end endgenerate
110endmodule

112// =======================================
113// Totally combinatorial count of number of bits set in a word
114// Uses recursion
115module two_set #(
116     parameter dw=8
117) (
118     input [dw-1:0] d,
119     output one,  // or more bits of d are set
120     output two   // or more bits of d are set
121);
122wire one_lower, one_upper;
123wire two_lower, two_upper;
124generate if (dw > 1) begin: split
125     two_set #(.dw(   dw/2)) lower(.d(d[dw/2-1  :0]), .one(one_lower), .two(two_lower));
126     two_set #(.dw(dw-dw/2)) upper(.d(d[dw-1 :dw/2]), .one(one_upper), .two(two_upper));
127     assign one = one_lower | one_upper;
128     assign two = two_lower | two_upper | (one_lower & one_upper);
129end else begin: passthrough
130     // single bit case is trivial
131     assign one = d[0];
132     assign two = 0;
133end endgenerate
134endmodule
