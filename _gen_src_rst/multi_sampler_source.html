 3/** MULTI SAMPLER
 4    Multi-purpose sampling signal generation with the following features:
 5    - Arbitrary base sampling period, defined as a fraction of input clock
 6      qualified with external trigger (ext_trig)
 7    - Main sampling signal
 8    - Three configurable downsampling strobes, defined w.r.t base sampling period
 9    - Downsampling strobe generation can be disabled to save HW if not required
10**/
11
12module multi_sampler #(
13   parameter sample_period_wi=8,
14   parameter dsample0_en=0,
15   parameter dsample0_wi=8,
16   parameter dsample1_en=0,
17   parameter dsample1_wi=8,
18   parameter dsample2_en=0,
19   parameter dsample2_wi=8)
20(
21   input                        clk,
22   input                        ext_trig,
23   input [sample_period_wi-1:0] sample_period,
24   input [dsample0_wi-1:0]      dsample0_period,
25   input [dsample1_wi-1:0]      dsample1_period,
26   input [dsample2_wi-1:0]      dsample2_period,
27   output                       sample_out,
28   output                       dsample0_stb,
29   output                       dsample1_stb,
30   output                       dsample2_stb
31);
32
33`ifdef SIMULATE
34   localparam INITVAL=1;
35`else
36   localparam INITVAL=0; // Longer startup
37`endif
38
39   reg [sample_period_wi-1:0] samp_per_r=0;
40   (* ASYNC_REG= "TRUE" *)   reg [sample_period_wi-1:0] base_count = 0;
41   (* ASYNC_REG= "TRUE" *)   reg [dsample0_wi-1:0]      ds0_count = INITVAL;
42   (* ASYNC_REG= "TRUE" *)   reg [dsample1_wi-1:0]      ds1_count = INITVAL;
43   (* ASYNC_REG= "TRUE" *)   reg [dsample2_wi-1:0]      ds2_count = INITVAL;
44   reg sample_out_l = 0;
45
46   // Base-timing generation
47   always @(posedge clk) begin
48      sample_out_l <= 0;
49      if (ext_trig) begin
50         samp_per_r <= sample_period;
51
52         base_count <= (base_count == (sample_period-1)) ? 0 : base_count+1;
53
54         // Restart count on sample_period change
55         if ((sample_period != samp_per_r) && |base_count) base_count <= 0;
56
57         sample_out_l <= (base_count == 0);
58      end
59   end
60   assign sample_out = sample_out_l;
61
62   generate if (dsample0_en) begin : g_dsample0
63      always @(posedge clk) begin
64         if (sample_out_l)
65            ds0_count <= (ds0_count == 1) ? dsample0_period : ds0_count-1;
66      end
67      assign dsample0_stb = ds0_count==1;
68   end else begin
69      assign dsample0_stb = 0;
70   end endgenerate
71
72   generate if (dsample1_en) begin : g_dsample1
73      always @(posedge clk) begin
74         if (sample_out_l)
75            ds1_count <= (ds1_count == 1) ? dsample1_period : ds1_count-1;
76      end
77      assign dsample1_stb = ds1_count==1;
78   end else begin
79      assign dsample1_stb = 0;
80   end endgenerate
81
82   generate if (dsample2_en) begin : g_dsample2
83      always @(posedge clk) begin
84         if (sample_out_l)
85            ds2_count <= (ds2_count == 1) ? dsample2_period : ds2_count-1;
86      end
87      assign dsample2_stb = ds2_count==1;
88   end else begin
89      assign dsample2_stb = 0;
90   end endgenerate
91
92endmodule
