 2// Dual port memory with independent clocks, port B is read-only
 3// Altera and Xilinx synthesis tools successfully "find" this as block memory
 4module dpram(
 5     clka, clkb,
 6     addra, douta, dina, wena,
 7     addrb, doutb
 8);
 9parameter aw=8;
10parameter dw=8;
11parameter sz=(32'b1<<aw)-1;
12parameter initial_file = "";
13
14input clka, clkb, wena;
15input [aw-1:0] addra, addrb;
16input [dw-1:0] dina;
17output [dw-1:0] douta, doutb;
18
19reg [dw-1:0] mem[sz:0];
20reg [aw-1:0] ala=0, alb=0;
21
22// In principle the zeroing loop should work OK for synthesis, but
23// there seems to be a bug in the Xilinx synthesizer
24// triggered when k briefly becomes sz+1.
25integer k=0;
26initial begin
27     if (initial_file != "") $readmemh(initial_file, mem);
28`ifdef SIMULATE
29     else begin
30             for (k=0;k<sz+1;k=k+1) mem[k]=0;
31     end
32`endif
33end
34
35assign douta = mem[ala];
36assign doutb = mem[alb];
37always @(posedge clka) begin
38     ala <= addra;
39     if (wena) mem[addra]<=dina;
40end
41always @(posedge clkb) begin
42     alb <= addrb;
43end
44
45endmodule
