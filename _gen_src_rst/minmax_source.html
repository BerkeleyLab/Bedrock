 4// reset is single-cycle, designed such that you can capture the outputs
 5// on the same cycle as you apply the reset, resulting in no blind cycles.
 6module minmax(clk,xin,reset,xmin,xmax);

14reg signed [width-1:0] xin_d=0;
15reg signed [width-1:0] xmin_r={width{1'b1}};
16reg signed [width-1:0] xmax_r={width{1'b0}};
17wire signed [width-1:0] max_plus = {1'b0,{(width-1){1'b1}}};
