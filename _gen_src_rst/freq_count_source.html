 1// Synthesizes to 86 slices at 312 MHz in XC3Sxxx-4 using XST-8.2i
 2//  (well, that's just the unknown frequency input; max sysclk is 132 MHz)
 3
 4`timescale 1ns / 1ns
 5
 6module freq_count #(
 7     // Default configuration useful for input frequencies < 96 MHz
 8     parameter glitch_thresh=2,
 9     parameter refcnt_width=24,
10     parameter freq_width=28,
11     parameter initv=0
12) (
13     // input clocks
14     input sysclk,  // timespec 8.0 ns
15     input f_in,  // unknown input
16
17     // outputs in sysclk domain
18     output reg [freq_width-1:0] frequency,
19     output freq_strobe,
20     output reg [15:0] diff_stream,
21     output reg diff_stream_strobe,
22     // glitch_catcher can be routed to a physical pin to trigger
23     // a 'scope; see glitch_thresh parameter above
24     output reg glitch_catcher
25);
26
27initial begin
28     frequency=initv;
29     diff_stream=0;
30     diff_stream_strobe=0;
31     glitch_catcher=0;
32end
33
34// four-bit Gray code counter on the input signal
35// http://en.wikipedia.org/wiki/Gray_code
36localparam gw=4;
37reg [gw-1:0] gray1=0;
38
39// The following three expressions compute the next Gray code based on
40// the current Gray code.  Vivado 2016.1, at least, is capable of reducing
41// them to the desired four LUTs when gw==4.
42// verilator lint_off UNOPTFLAT
43wire [gw-1:0] bin1 = gray1 ^ {1'b0, bin1[gw-1:1]};  // Gray to binary
44// verilator lint_on UNOPTFLAT
45wire [gw-1:0] bin2 = bin1 + 1;  // add one
46wire [gw-1:0] gray_next = bin2 ^ {1'b0, bin2[gw-1:1]};  // binary to Gray
47always @(posedge f_in) gray1 <= gray_next;
48
49// transfer that Gray code to the measurement clock domain
50reg [gw-1:0] gray2=0, gray3=0;
51always @(posedge sysclk) begin
52     gray2 <= gray1;
53     gray3 <= gray2;
54end
55
56// verilator lint_off UNOPTFLAT
57wire [gw-1:0] bin3 = gray3 ^ {1'b0, bin3[gw-1:1]}; // convert Gray to binary
58// verilator lint_on UNOPTFLAT
59
60reg [gw-1:0] bin4=0, bin5=0, diff1=0;
61always @(posedge sysclk) begin
62     bin4 <= bin3;
63     bin5 <= bin4;
64     diff1 <= bin4-bin5;
65     if (diff1 > glitch_thresh) glitch_catcher <= ~glitch_catcher;
66end
67
68// It might also be possible to histogram diff1,
69// but for now just accumulate it to get a traditional frequency counter.
70// Also make it available to stream to host at 24 MByte/sec, might be
71// especially interesting when reprogramming a clock divider like
72// the AD9512 on a LLRF4 board.
73// In that case a 48 MHz sysclk / 2^24 = 2.861 Hz update
74reg [freq_width-1:0] accum=0, result=initv;
75reg [refcnt_width-1:0] refcnt=0;
76reg ref_carry=0;
77reg [15:0] stream=0;
78reg stream_strobe=0;
79always @(posedge sysclk) begin
80     {ref_carry, refcnt} <= refcnt + 1;
81     if (ref_carry) result <= accum;
82     accum <= (ref_carry ? 28'b0 : accum) + diff1;
83     stream <= {stream[11:0],diff1};
84     stream_strobe <= refcnt[1:0] == 0;
85end
86
87// Latch/pipeline one more time to perimeter of this module
88// to make routing easier
89reg freq_strobe_r=0;
90always @(posedge sysclk) begin
91     frequency <= result;
92     freq_strobe_r <= ref_carry;
93     diff_stream <= stream;
94     diff_stream_strobe <= stream_strobe;
95end
96assign freq_strobe = freq_strobe_r;
97
98endmodule
