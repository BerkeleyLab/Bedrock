 1// Simple first-order CIC filter and decimator
 2// Note that this module is configured with unsigned input and output!
 3module cic_simple_us(
 4     input clk,
 5     input [dw-1:0] data_in,
 6     input data_in_gate,
 7     input roll,  // sometimes unused, see ext_roll parameter
 8     output [dw-1:0] data_out,
 9     output data_out_gate
10);
11parameter ext_roll=0;  // if set, use roll port instead of internal divider
12parameter dw=16;
13parameter ex=10;  // decimate by 2^ex, up to 2^ex when using ext_roll
14
15reg [dw+ex-1:0] data_int=0, data_int_h=0, diff=0;
16reg [ex-1:0] div=0;
17reg iroll=0, roll_r=0;  // only one of these will be used
18always @(posedge clk) if (data_in_gate) begin
19     data_int <= data_int + data_in;  // unsigned arithmetic, zero-padding for data_in
20     {iroll, div} <= div + 1;
21     roll_r <= roll;  // like data_in, roll is sampled on the cycle marked by data_in_gate
22end
23wire uroll = ext_roll ? roll_r : iroll;
24
25reg data_in_gate2=0, data_out_gate_r=0;
26wire deriv_gate = data_in_gate2 & uroll;
27always @(posedge clk) begin
28     data_in_gate2 <= data_in_gate;
29     data_out_gate_r <= deriv_gate;
30     if (deriv_gate) begin
31             diff <= data_int - data_int_h;
32             data_int_h <= data_int;
33     end
34end
35
36assign data_out_gate = data_out_gate_r;
37assign data_out = diff[dw+ex-1:ex];
38
39endmodule
