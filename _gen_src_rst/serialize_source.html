 3// Encapsulation of the serialization step in LBNL's conveyor belt
 4// Instantiated by e.g., serializer_multichannel.v
 5module serialize(
 6     input clk,  // timespec 8.4 ns
 7     input samp, // Snap signal for data_in
 8     input signed [dwi-1:0] data_in,
 9     input signed [dwi-1:0] stream_in,
10     output signed [dwi-1:0] stream_out,
11     input gate_in,
12     output gate_out,
13     output strobe_out
14);
15parameter dwi=28;  // result width
16// Difference between above two widths should be N*log2 of the maximum number
17// of samples per CIC sample, where N=2 is the order of the CIC filter.
18
19reg signed [dwi-1:0] stream_reg=0;
20reg gate_reg=0;
21always @(posedge clk) begin
22     stream_reg <= samp ? data_in : stream_in;
23     gate_reg <= samp ? samp : gate_in;
24end
25assign stream_out = stream_reg;
26assign gate_out = gate_reg;
27assign strobe_out=samp;
28endmodule
