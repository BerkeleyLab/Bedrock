 3// DMTD measurement of clock phasing
 4module phase_diff #(
 5     // Default parameters tuned for LCLS-II LLRF Digitizer,
 6     // where unknown clocks are 94.286 MHz and sampling clock is 200 MHz.
 7     parameter order1=1,
 8     parameter order2=1,
 9     parameter dw=14,
10     parameter adv=3861,
11     parameter delta=33,
12     parameter ext_div1_en=0,
13     parameter ext_div2_en=0
14) (
15     input uclk1,  // unknown clock 1
16     input ext_div1, // external divider on uclk1
17     input uclk2,  // unknown clock 2
18     input ext_div2, // external divider on uclk2
19     input sclk,   // sampling clock
20     input rclk,   // readout clock (data transfer, local bus)
21     // the following are in rclk domain
22     output dval,
23     output [dw-2:0] phdiff_out,
24     output [dw-1:0] vfreq_out,
25     output locked
26);
27
28// XXX vfreq_out could have extra msbs added by unwrapping, or better still,
29// adding higher-order bits to phaset.
30
31// Two phase trackers
32wire [dw-1:0] phaset_out1, phaset_out2;
33wire fault1, fault2;
34phaset #(
35     .ext_div_en(ext_div1_en),
36     .order(order1), .dw(dw), .adv(adv), .delta(delta)
37) track1(
38     .uclk(uclk1),
39     .sclk(sclk),
40     .ext_div(ext_div1),
41     .phase(phaset_out1),
42     .fault(fault1));
43
44phaset #(
45     .ext_div_en(ext_div2_en),
46     .order(order2), .dw(dw), .adv(adv), .delta(delta)
47) track2 (
48     .uclk(uclk2),
49     .sclk(sclk),
50     .ext_div(ext_div2),
51     .phase(phaset_out2),
52     .fault(fault2));
53
54// Don't assume sclk phase is stable, just report the difference.
55// msb of phaset_out* is useless, just represents internal divider state;
56// that msb will be dropped in the pass through data_xdomain below.
57reg [dw-1:0] ph_diff_sclk=0;
58reg [dw-1:0] ph_sum=0, ph_sum_old=0, vernier_freq=0;
59reg [10:0] cnt=0;
60wire tick = &cnt;
61wire fault = fault1 | fault2;
62reg err_r=1;
63always @(posedge sclk) begin
64     ph_diff_sclk <= phaset_out1 - phaset_out2;
65     ph_sum <= phaset_out1 + phaset_out2;
66     cnt <= cnt+1;
67     if (tick | fault) err_r <= fault;
68     if (tick & locked) begin
69             ph_sum_old <= ph_sum;
70             vernier_freq <= ph_sum - ph_sum_old;
71     end
72end
73assign locked = ~err_r;  // no fancy clock domain crossing
74
75// Periodically pass the result to rclk domain
76wire din_stb = &cnt[4:0] && locked;
77data_xdomain #(.size(dw+dw-1)) xdom(
78     .clk_in(sclk), .gate_in(din_stb), .data_in({ph_diff_sclk[dw-2:0], vernier_freq}),
79     .clk_out(rclk),.gate_out(dval), .data_out({phdiff_out, vfreq_out}));
80
81endmodule
