 1// Synthesizes to 251 slices at 116 MHz in XC3Sxxx-4 using XST-8.2i
 2`timescale 1ns / 1ns
 3
 4module half_filt(
 5     input clk,  // Rising edge clock input; all logic is synchronous in this domain
 6     input signed [19:0] ind,  // Input data
 7     input ing,  // Active high gate marking input data as valid
 8     output signed [19:0] outd,  // Output data
 9     output outg,  // Active high gate marking output data as valid
10     input reset  // manually reset counter
11);
12
13parameter len = 4;  // number of interleaved data streams
14wire signed [19:0] d0, d1, d2, d3, d4, d5, d6, d7, d8, d9, d10;
15
16assign d0 = ind;
17reg_delay #(.dw(20),.len(len)) h0(clk, 1'b0, ing, d0, d1);
18reg_delay #(.dw(20),.len(len)) h1(clk, 1'b0, ing, d1, d2);
19reg_delay #(.dw(20),.len(len)) h2(clk, 1'b0, ing, d2, d3);
20reg_delay #(.dw(20),.len(len)) h3(clk, 1'b0, ing, d3, d4);
21reg_delay #(.dw(20),.len(len)) h4(clk, 1'b0, ing, d4, d5);
22reg_delay #(.dw(20),.len(len)) h5(clk, 1'b0, ing, d5, d6);
23reg_delay #(.dw(20),.len(len)) h6(clk, 1'b0, ing, d6, d7);
24reg_delay #(.dw(20),.len(len)) h7(clk, 1'b0, ing, d7, d8);
25reg_delay #(.dw(20),.len(len)) h8(clk, 1'b0, ing, d8, d9);
26reg_delay #(.dw(20),.len(len)) h9(clk, 1'b0, ing, d9, d10);
27
28reg signed [20:0] s1=0, s2=0, s3=0, s4=0;
29reg sg=0;
30always @(posedge clk) begin
31     s1 <= d0 + d10;
32     s2 <= d2 + d8;
33     s3 <= d4 + d6;
34     s4 <= d5 + 0; // {d5[19],d5};
35     sg <= ing;
36end
37
38// FIR filter coefficients:
39// 2*(1/32 0 -1/8-1/64 0 1/2+1/8-1/64 1 1/2+1/8-1/64 0 -1/8-1/64 0 1/32)/4
40// see lowp3.m
41
42reg signed [21:0] a1=0, a2=0, a3=0, a4=0;
43reg ag=0;
44always @(posedge clk) begin
45`ifdef TRUST_VERILOG_DIVISION
46     a1 <= s1/16 + 0;
47     a2 <= s2/4 + s2/32;
48     a3 <= s3/4 - s3/32;
49     a4 <= s3 + 2*s4;
50`else
51     a1 <= {{5{s1[20]}},s1[20:4]};
52     a2 <= {{3{s2[20]}},s2[20:2]} + {{6{s2[20]}},s2[20:5]};
53     a3 <= {{3{s3[20]}},s3[20:2]} - {{6{s3[20]}},s3[20:5]};
54     a4 <= {{1{s3[20]}},s3[20:0]} + {            s4[20:0],1'b0};
55`endif
56     ag <= sg;
57end
58
59reg signed [22:0] b1=0, b2=0;
60reg bg=0;
61reg [8:0] samp=0;
62reg show=0;
63always @(posedge clk) begin
64     b1 <= a1 - a2;
65     b2 <= a3 + a4 + 1;
66     bg <= ag;
67     if (reset) begin
68             samp <= 0;
69             show <= 0;
70     end
71     else begin
72             if (bg) begin
73                     samp <= (samp==len-1'b1) ? 0 : samp+1'b1;
74                     if (samp==len-1) show <= ~show;
75             end
76     end
77end
78
79wire signed [21:0] c1;
80sat_add #(23,22) finals(clk,b1,b2,c1);
81reg cg=0;
82always @(posedge clk) cg <= bg & show;
83
84assign outd = c1[21:2];
85assign outg = cg;
86
87endmodule
