 1//
 2// Simple-minded integer square root
 3//
 4// General purpose with the caveat that XWIDTH should be even.
 5// X_WIDTH/2 + 1 cycles per operation, non-pipelined.
 6// Setting en when computation is in progress will
 7// cancel the computation and start another.
 8//
 9// With XWIDTH=32, synthesizable to 110 MHz on Spartan3-5.
10//
11// Eric Norum
12// LBNL
13//
14
15module isqrt(clk, x, en, y, dav);
16
17parameter X_WIDTH     = 32;
18parameter Y_WIDTH    = X_WIDTH / 2;
19
20input                 clk;
21input  [X_WIDTH-1:0]  x;
22input                 en;
23output [Y_WIDTH-1:0]  y;
24output                dav;
25
26wire [Y_WIDTH-1:0]  y;
27reg  [X_WIDTH-1:0]  op = 0, pw4 = 0, res = 0;
28reg                 dav = 0;
29
30assign y = res[Y_WIDTH-1:0];
31
32always @(posedge clk)
33begin
34    if (en) begin
35        //
36        //  op = x
37        // res = 0
38        // pw4 = highest power of 4 less than largest X
39        //
40        op <= x;
41        res <= 0;
42        pw4 <= {1'b1 ,{Y_WIDTH-1{2'b0}}};
43        dav <= 0;
44    end else begin
45        //
46        // while (pw4 != 0) {
47        //     if (op >= res + pw4) {
48        //         op = op - (res + pw4);
49        //         res = res +  2 * pw4;
50        //     }
51        //     res >>= 1;
52        //     pw4 >>= 2;
53        // }
54        //
55        if (pw4 != 0) begin
56            if (op >= (res + pw4)) begin
57                op <= op - (res + pw4);
58                res <= (res >> 1) + pw4;
59            end else begin
60                res <= (res >> 1);
61            end
62            pw4 <= (pw4 >> 2);
63        end
64        if (pw4 == 1) begin
65            dav <= 1;
66        end else begin
67            dav <= 0;
68        end
69    end
70end
71endmodule
