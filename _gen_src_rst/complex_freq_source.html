  1module complex_freq #(
  2     parameter refcnt_w = 17
  3) (
  4     input clk,  // single clock domain
  5     input signed [17:0] sdata,
  6     input sgate,  // high for two cycles representing I and Q
  7     output signed [refcnt_w-1:0] freq,
  8     output freq_valid, // Asserted when freq output is valid
  9     output [16:0] amp_max,
 10     output [16:0] amp_min,
 11     output updated, // Asserted when amp_{max,min} are updated
 12     output timing_err, // New data received while calculation is ongoing
 13     //
 14     // Additional output giving magnitude-squared, meant to
 15     // support average power calculations; can be ignored.
 16     output [23:0] square_sum_out,
 17     output square_sum_valid
 18);
 19
 20// One multiplier to square the inputs
 21reg signed [35:0] square, square_d;
 22reg signed [36:0] square_sum;
 23always @(posedge clk) begin
 24     square <= sdata*sdata;
 25     square_d <= square;
 26     square_sum <= square + square_d;
 27end
 28wire [33:0] square_sum_cut = square_sum[33:0];
 29
 30// Find cycle of valid sum
 31reg sgate1, sgate2, sum_valid;
 32always @(posedge clk) begin
 33     sgate1 <= sgate;
 34     sgate2 <= sgate1;
 35     sum_valid <= sgate1 & sgate2;
 36end
 37
 38assign square_sum_out = square_sum_cut[33:10];
 39assign square_sum_valid = sum_valid;
 40
 41// Sqrt for convenience and to keep the word width down
 42wire [16:0] sqrt_val;
 43wire sqrt_valid;
 44isqrt #(.X_WIDTH(34)) sqrt(.clk(clk), .x(square_sum_cut), .en(sum_valid),
 45     .y(sqrt_val), .dav(sqrt_valid));
 46
 47// Catch obvious errors; leave latching to the upper layers
 48reg busy=0, timing_err_r=0;
 49always @(posedge clk) begin
 50     if (sum_valid) busy <= 1;
 51     if (sqrt_valid) busy <= 0;
 52     timing_err_r <= busy & sgate;
 53end
 54assign timing_err = timing_err_r;
 55
 56// Find min and max
 57wire rollover;
 58reg [16:0] amp_max_x, amp_min_x;  // developing values
 59reg [16:0] amp_max_r, amp_min_r;  // frozen values reported to caller
 60wire cmp_gt = sqrt_val > amp_max_x;
 61wire cmp_lt = sqrt_val < amp_min_x;
 62always @(posedge clk) begin
 63     if (sqrt_valid && (cmp_gt || rollover)) amp_max_x <= sqrt_val;
 64     if (sqrt_valid && (cmp_lt || rollover)) amp_min_x <= sqrt_val;
 65     if (sqrt_valid && rollover) begin
 66             amp_max_r <= amp_max_x;
 67             amp_min_r <= amp_min_x;
 68     end
 69end
 70assign amp_max = amp_max_r;
 71assign amp_min = amp_min_r;
 72
 73// Reference counter
 74reg [refcnt_w-1:0] refcnt=1;
 75reg rollover_r=0;
 76always @(posedge clk) begin
 77     if (sqrt_valid) refcnt <= refcnt-1;
 78     rollover_r <= &refcnt;
 79end
 80assign rollover = rollover_r;
 81
 82// Frequency counter
 83reg [refcnt_w:0] quad_cnt, freq_r;
 84reg [1:0] quad, oldquad;
 85wire [1:0] transition = quad - oldquad;
 86reg invalid=0, freq_valid_r=0;
 87wire newbit = sdata[17] ^ (sgate1 & quad[0]);
 88reg updated_r=0;
 89always @(posedge clk) begin
 90     if (sgate) quad <= {quad[0], newbit};
 91     if (sum_valid) begin
 92             oldquad <= quad;
 93             case (transition)
 94             0: quad_cnt <= quad_cnt;
 95             1: quad_cnt <= quad_cnt-1;
 96             2: invalid <= 1;
 97             3: quad_cnt <= quad_cnt+1;
 98             endcase
 99     end
100     if (sqrt_valid && rollover) begin
101             freq_r <= quad_cnt;
102             freq_valid_r <= ~invalid;
103             quad_cnt <= 0;
104             invalid <= 0;
105     end
106     updated_r <= sqrt_valid && rollover;
107end
108assign freq = freq_r[refcnt_w:1];
109assign freq_valid = freq_valid_r;
110assign updated = updated_r;
111
112endmodule
