 1// Digital phase-locked-loop tracker
 2
 3// At least in this application, the sampling clock is not phase-locked
 4// to the clocks of interest.  So a binary phase accumulator is as good
 5// as any, and it makes the arithmetic for subtracting the two results
 6// (from the two unknown clocks) easy.
 7
 8`timescale 1ns / 1ns
 9
10module phaset #(
11     parameter order=1,
12     parameter dw=14,
13     parameter adv=3861,
14     parameter delta=264,  // 8*33
15     parameter ext_div_en=0
16) (
17     input uclk,
18     input sclk,
19     input ext_div,
20     output [dw-1:0] phase,
21     output fault  // single cycle
22);
23
24// _Still_ draw analogy to the AD9901
25// Generalize to a Johnson counter
26reg [order-1:0] ishr=0;
27always @(posedge uclk) ishr <= (ishr << 1) | {{order-1{1'b0}},~ishr[order-1]};
28wire div = ishr[0];
29//reg div=0; always @(posedge uclk) div <= ~div;
30
31// Test bench fails for some initial phase_r values between 14900 and 15050.
32// In that case the fault output signals the problem.
33reg [dw-1:0] osc=0, acc=0, phase_r=0;
34reg capture=0;
35reg fault_r=0;
36wire msb = acc[dw-1];
37wire nsb = acc[dw-2];
38wire peak = acc[dw-2] ^ acc[dw-3];  // peak of virtual sine wave
39wire move = capture != msb;
40wire dir  = ~nsb;
41wire dn = move &  dir;
42wire up = move & ~dir;
43always @(posedge sclk) begin
44     capture <= ext_div_en ? ext_div : div;
45     if (move) phase_r <= phase_r + (dir ? -delta : delta);
46     fault_r <= move & peak;
47     osc <= osc + adv;
48     acc <= osc + phase_r;
49end
50assign phase = phase_r;
51assign fault = fault_r;
52
53endmodule
