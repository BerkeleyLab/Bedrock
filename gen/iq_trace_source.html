  1// One more try to package up the LBNL conveyor belt in a usable and
  2// comprehensible form.
  3
  4// New things going on here:
  5//  1. adc count is parameterized, and therefore a single input port
  6//     is nadc*dw bits wide; each block of dw bits is a signed number.
  7//  2. Independent I and Q shift registers
  8//  3. 32-bit wide memory read port can either hold the 20-bit full-resolution
  9//     measurement, or two packed 16-bit I and Q data
 10
 11// LCLS-II dissection of possible use cases
 12// 1320/14 MHz raw ADC clock
 13// /33 for base CIC, 2.857 MS/s
 14// /8 to get 357 kS/s, suitable for loop characterization
 15// or /16 to get 179 kS/s, and a channel-subset is streamable for close-in
 16//   noise characterization
 17//   179 kS/s * 2 ADCs * 2 components * 20 bits = 14.3 Mbit/s
 18//   or without re-writing any infrastructure now,
 19//   179 kS/s * 2 ADCs * 2 components * 64 bits = 45.7 Mbit/s
 20// or /64 to get 44.64 kS/s, easily streamable and good for acoustic
 21//   characterization, including most SRF trip analysis
 22//   44.64 kS/s * 8 cavities * 2 components * 16 bits = 11.4 Mbit/s
 23//   Useful to grab 8 packets at a time with 1024 octets each,
 24//   requires each half of a ping-pong buffer to be 4K x 16.
 25
 26// To support more than 33*16-point averaging, without losing resolution, you
 27// need wider data paths in ccfilt, with its recently-added outw parameter.
 28// 16 + log2(33*16/2)/2 = 20
 29// where the /2 inside the log represents mean(sin(theta)^2), and the answer
 30// comes out slightly more than 20 because we've ignored the intended
 31// adjustment to lo_amp.
 32
 33// See nco_setup.py for help setting phase_step_h, phase_step_l, and modulo.
 34
 35module iq_trace #(
 36   parameter dw         = 16, // ADC input width
 37   parameter oscw       = 18, // Oscillator input width
 38   parameter davr       = 3,  // Guard bits at output of mixer
 39   parameter ow         = 28, // second-order CIC data path width
 40   parameter rw         = 20, // result width out of ccfilt
 41   parameter pcw        = 13, // cic_period counter width
 42   parameter shift_base = 7,  // see ccfilt.v
 43   parameter nadc       = 8,
 44   parameter aw         = 13  // for circle_buf, see below
 45) (
 46  input                      clk,
 47  input                      reset,
 48  input                      trig,
 49  input        [1:0]         trig_mode,  // 0: free-run, 1: single-shot, 2: sync start, XXX not yet used
 50  input        [nadc*dw-1:0] adcs,   // each dw-wide adc data is signed
 51  input signed [oscw-1:0]    cosa,
 52  input signed [oscw-1:0]    sina,
 53  // Presumably host-settable parameters in the clk domain
 54  input        [pcw-1:0]     cic_period,  // expected values 33 to 33*128
 55  input        [3:0]         cic_shift,   // expected values 7 to 15
 56  input        [nadc-1:0]    keep,   // bit n :: channel n
 57  // 18+10+4 = 32 so all of the last three parameters could be a single
 58  // configuration word that fully defines output scaling
 59  //
 60  // host readout port, based on circle_buf.  ro_data is:
 61  //  ro_addr[aw+1] == 1'b1       packed 16-bit {I, Q} results
 62  //  ro_addr[aw+1:aw] == 2'b10   long (rw bits) I result
 63  //  ro_addr[aw+1:aw] == 2'b11   long (rw bits) Q result
 64  input                      ro_clk,
 65  output                     ro_enable,
 66  input                      ro_ack,
 67  input  [aw+1:0]            ro_addr,
 68  output [31:0]              ro_data,
 69  output [31:0]              ro_status
 70);
 71
 72   // Set up some global signals in the right clock domain
 73   reg [1:0] trig_mode_r = 0;
 74   reg       trig_pending = 0;
 75   wire      boundary;
 76
 77   wire trig_actual = trig_pending & boundary;
 78   always @(posedge clk) begin
 79      if (boundary) trig_pending <= 0;
 80      if (trig)     trig_pending <= 1;
 81
 82      trig_mode_r <= trig_mode;
 83   end
 84
 85   // ---------------------
 86   // Instantiate Sampler
 87   // ---------------------
 88   wire cic_sample;
 89
 90   multi_sampler #(
 91      .sample_period_wi (pcw))
 92   i_multi_sampler (
 93      .clk             (clk),
 94      .ext_trig        (1'b1),
 95      .sample_period   (cic_period),
 96      .dsample0_period (8'h1),
 97      .dsample1_period (8'h1),
 98      .dsample2_period (8'h1),
 99      .sample_out      (cic_sample),
100      .dsample0_stb    (),
101      .dsample1_stb    ()
102   );
103
104   // ---------------------
105   // Instantiate mixers to create I and Q streams
106   // ---------------------
107   wire signed [nadc*(dw+davr)-1:0] mixout_i, mixout_q;
108
109   iq_mixer_multichannel #(
110      .NCHAN (nadc),
111      .DWI   (dw),
112      .DAVR  (davr),
113      .DWLO  (oscw)
114   ) i_iq_mixer_multichannel (
115      .clk      (clk),
116      .adc      (adcs),
117      .cos      (cosa),
118      .sin      (sina),
119      .mixout_i (mixout_i),
120      .mixout_q (mixout_q)
121   );
122
123   // ---------------------
124   // Instantiate two CIC_MULTICHANNEL, one for each stream
125   // ---------------------
126   wire strobe_cc;
127   wire signed [rw-1:0] result_i, result_q;
128
129   cic_multichannel #(
130      .n_chan        (nadc),
131      // DI parameters
132      .di_dwi        (dw+davr),
133      .di_rwi        (ow),
134      .di_noise_bits (1), // NOTE: Setting to 1 to compensate for removed /2 from double_inte
135      .cc_outw       (rw),
136      .cc_halfband   (0),
137      .cc_use_delay  (0),
138      .cc_shift_base (shift_base))
139   i_cic_multichannel_i
140   (
141      .clk           (clk),
142      .reset         (reset),
143      .stb_in        (1'b1),
144      .d_in          (mixout_i),
145      .cic_sample    (cic_sample),
146      .cc_sample     (1'b1),
147      .cc_shift      (cic_shift),
148      .di_stb_out    (), // Unused double-integrator tap
149      .di_sr_out     (),
150      .cc_stb_out    (strobe_cc),
151      .cc_sr_out     (result_i)
152   );
153
154   cic_multichannel #(
155      .n_chan        (nadc),
156      // DI parameters
157      .di_dwi        (dw+davr),
158      .di_rwi        (ow),
159      .di_noise_bits (1), // NOTE: Setting to 1 to compensate for removed /2 from double_inte
160      .cc_outw       (rw),
161      .cc_halfband   (0),
162      .cc_use_delay  (0),
163      .cc_shift_base (shift_base))
164   i_cic_multichannel_q
165   (
166      .clk           (clk),
167      .reset         (reset),
168      .stb_in        (1'b1),
169      .d_in          (mixout_q),
170      .cic_sample    (cic_sample),
171      .cc_sample     (1'b1),
172      .cc_shift      (cic_shift),
173      .di_stb_out    (),
174      .di_sr_out     (),
175      .cc_stb_out    (), // Time-aligned with strobe_cc
176      .cc_sr_out     (result_q)
177   );
178
179
180   // ---------------------
181   // Instantiate circle_buf_serial to store I and Q streams and handle channel masking
182   // ---------------------
183
184   assign boundary = ~strobe_cc;
185
186   // Run/stop mode based on trigger and trigger mode
187   wire buf_sync;
188   reg run=0;
189   always @(posedge clk) begin
190      if (trig_mode_r == 0)                run <= 1;
191      if (trig_mode_r == 1 && buf_sync)    run <= 0;
192      if (trig_mode_r != 0 && trig_actual) run <= 1;
193   end
194   wire circle_stb = strobe_cc & run;
195
196   // Circular buffer
197   // 2 * 20 bits from 2 * ccfilt, 8K depth * 2 I&Q channels means aw=13
198   // 2 * 8K * 2 * 20 = 640 kbits = 20 BRAM36 in Xilinx 7-series
199   wire [2*rw-1:0] circle_out;
200
201   circle_buf_serial #(
202      .n_chan        (nadc),
203      .lsb_mask      (1), // Right to left. LSB=CH0
204      .buf_aw        (aw),
205      .buf_dw        (2*rw),
206      .buf_auto_flip (0))
207   i_circle_buf_serial (
208      .iclk            (clk),
209      .sr_in           ({result_q,result_i}),
210      .sr_stb          (circle_stb),
211      .chan_mask       (keep),
212      .oclk            (ro_clk),
213      .buf_sync        (buf_sync),
214      .buf_transferred (),
215      .buf_stop        (1'b0),
216      .buf_count       (ro_status[31:16]),
217      .buf_stat2       (),
218      .buf_stat        (ro_status[15:0]),
219      .debug_stat      (),
220      .stb_out         (ro_ack),
221      .enable          (ro_enable),
222      .read_addr       (ro_addr[aw-1:0]),
223      .d_out           (circle_out)
224   );
225
226   // Form 32-bit output bus
227   reg [31:0] obus;
228   wire [1:0] obus_mode = ro_addr[aw+1:aw];
229   always @(*) casez (obus_mode)
230           2'b0?: obus = {circle_out[rw-1:rw-16], circle_out[2*rw-1:2*rw-16]};
231           2'b10: obus = {circle_out[  rw-1:0],  {(32-rw){1'b0}}};
232           2'b11: obus = {circle_out[2*rw-1:rw], {(32-rw){1'b0}}};
233   endcase
234   assign ro_data = obus;
235
236endmodule
